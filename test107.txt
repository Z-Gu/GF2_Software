DEVICES	AND A1 2,
	XOR X1,
	XOR X2,
	XOR X3,
	D_TYPE D1,
	D_TYPE D2,
	D_TYPE D3,
	CLOCK CLK1 5,
	SWITCH I 1,
	SWITCH Reset 0,
	SWITCH Set 0;
CONNECT	I => X1.I2,
	D1.Q => X1.I1,
	CLK1 => D1.CLK,
	X1 => D1.DATA,
	D1.Q => A1.I1,
	D1.Q => X2.I2,
	D2.Q => X2.I1,
	X2 => D2.DATA,
	CLK1 => D2.CLK,
	D2.Q => A1.I2,
	A1 => X3.I1,
	D3.Q => X3.I2,
	X3 => D3.DATA,
	CLK1 => D3.CLK,
	Reset => D1.CLEAR,
	Reset => D2.CLEAR,
	Reset => D3.CLEAR,
	Set => D1.SET,
	Set => D2.SET,
	Set => D3.SET;
MONITOR	D3.Q;
